ObjectT WorldModel::getTeammateAtAngleEx(AngDeg angA, AngDeg angB, double dDist)
{
    VecPosition posAgent = getAgentGlobalPosition();
    VecPosition posTeammate;
    AngDeg angTeammate;
    int iIndex;
    
    for(ObjectT o = iterateObjectStart(iIndex, OBJECT_SET_TEAMMATES);o != OBJECT_ILLEGAL;o = iterateObjectNext(iIndex, OBJECT_SET_TEAMMATES) )
    {
        posTeammate = getGlobalPosition(o);    
        angTeammate = (posTeammate - posAgent).getDirection();
        if(angA <= angTeammate && angTeammate <= angB && posAgent.getDistanceTo(posTeammate) < dDist)
        {
             return o;
        }
    }
    iterateObjectDone(iIndex);
    return OBJECT_ILLEGAL;
}

ObjectT WorldModel::getTeammate()
{
    VecPosition posAgent = getAgentGlobalPosition();//当前球员位置
    VecPosition posTeammate;//对手位置
    int iIndex;
    int iIndex2;
    ObjectT oBest = getAgentObjectType();
    VecPosition posBest = posAgent;
    
    Rect recUp(OBJECT_FLAG_L_T,OBJECT_FLAG_R_0);
    Rect recDown(OBJECT_FLAG_L_0,OBJECT_FLAG_R_B);
    if(getNrInSetInRectangle(OBJECT_SET_OPPONENTS,&recDown) > getNrInSetInRectangle(OBJECT_SET_OPPONENTS,&recUp))
    {
        iIndex2 = 1; 
    }
    else
    {
        iIndex2 = -1;
    }
    
    for(ObjectT o = iterateObjectStart(iIndex, OBJECT_SET_TEAMMATES);o != OBJECT_ILLEGAL;o = iterateObjectNext(iIndex, OBJECT_SET_TEAMMATES) )
    {
        posTeammate = getGlobalPosition(o);   
        if(iIndex2 == 1 && posTeammate.getY() > 0)
        {
            if(posTeammate.getX() > posBest.getX())
            {
                oBest = o;
                posBest = getGlobalPosition(oBest);
            }
        }
        else if(iIndex2 == -1 && posTeammate.getY() < 0)
        {
            if(posTeammate.getX() > posBest.getX())
            {
                oBest = o;
                posBest = getGlobalPosition(oBest);
            }
        }
    }
    iterateObjectDone(iIndex);
    return oBest;
}

ObjectT WorldModel::getTeammateCloseToBall()
{
    VecPosition posBall = getBallPos();
    
    ObjectT oSet[] = {OBJECT_TEAMMATE_2, OBJECT_TEAMMATE_3, OBJECT_TEAMMATE_4};
    ObjectT oCloseset = oSet[0];
    
    ObjectT oTemp;
    double distCloset = getGlobalPosition(oCloseset).getDistanceTo(posBall);
    
    for(int i = 1; i < 3; i++)
    {
        oTemp = oSet[i];
        if(getGlobalPosition(oTemp).getDistanceTo(posBall) < distCloset)
        {
            oCloseset = oTemp;
        }
    }
    
    return oCloseset;
}
